$Sun = [Int] [DayOfWeek] "Sunday"
$Sat = [Int] [DayOfWeek] "Saturday"

$now = Get-Date

if ($min.TimeOfDay -le $now.TimeOfDay -and $max.TimeOfDay -ge $now.TimeOfDay -and $Sun -le $now.DayOfWeek -and $Sat -ge $now.DayOfWeek) {
  $StartingArgs.CancelStartup = $true
}
